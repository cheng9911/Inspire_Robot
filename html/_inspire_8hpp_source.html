181std::string Inspire::GetCanId(int Address, int handid, int flag)
182{
183 std::string AddressId = GetAddressId(Address);
184 std::string HandId = GetHandId(handid);
185 std::string flag_write_read = GetFlageWriteRead(flag);
186
187 // 检查 AddressId 和 HandId 是否为空
188 if (AddressId.size() == 0 || HandId.size() == 0 || flag_write_read.size() == 0)
189 {
190 std::cout << "AddressId or HandId or flag_write_read is empty" << std::endl;
191 return "";
192 }
193 // 检查 AddressId 和 HandId 是否为 12 位和 14 位
194 if (AddressId.size() != 12 || HandId.size() != 14 || flag_write_read.size() != 3)
195 {
196 std::cout << "AddressId or HandId or flag_write_read is not 12 or 14 or 3 bits " << std::endl;
197 return "";
198 }
199
200 std::string tmp = "000";
201 // 拼接 AddressId 和 HandId
202 std::string CanId = tmp + flag_write_read + AddressId + HandId;
203 return CanId;
204}
205void Inspire::ConvertValueToHex(int value, Byte data[2])
206{
207 // 检查 value 是否在范围内
208 if (value < 0 || value > 1000)
209 {
210 std::cerr << "Error: Value out of range [0, 1000]" << std::endl;
211 return;
212 }
213
214 // 将 value 转换为十六进制并存储到 data 数组中
215 std::stringstream ss;
216 ss << std::hex << std::uppercase << std::setw(4) << std::setfill('0') << value;
217 std::string hexString = ss.str();
218
219 // 检查是否需要补充 0，确保有 4 位
220 if (hexString.length() < 4)
221 {
222 hexString = std::string(4 - hexString.length(), '0') + hexString;
223 }
224
225 // 交换高低字节
226 data[0] = std::stoi(hexString.substr(2, 2), nullptr, 16); // 高字节
227 data[1] = std::stoi(hexString.substr(0, 2), nullptr, 16); // 低字节
228}
