164 std::cout << "it is not OpenDevice" << std::endl;

238std::string Inspire::GetCanId(int Address, int handid, int flag)
239{
240 std::string AddressId = GetAddressId(Address);
241 std::string HandId = GetHandId(handid);
242 std::string flag_write_read = GetFlageWriteRead(flag);
243
244 // 检查 AddressId 和 HandId 是否为空
245 if (AddressId.size() == 0 || HandId.size() == 0 || flag_write_read.size() == 0)
246 {
247 std::cout << "AddressId or HandId or flag_write_read is empty" << std::endl;
248 return "";
249 }
250 // 检查 AddressId 和 HandId 是否为 12 位和 14 位
251 if (AddressId.size() != 12 || HandId.size() != 14 || flag_write_read.size() != 3)
252 {
253 std::cout << "AddressId or HandId or flag_write_read is not 12 or 14 or 3 bits " << std::endl;
254 return "";
255 }
256
257 std::string tmp = "000";
258 // 拼接 AddressId 和 HandId
259 std::string CanId = tmp + flag_write_read + AddressId + HandId;
260 return CanId;

262void Inspire::ConvertValueToHex(int value, Byte data[2])
263{
264 // 检查 value 是否在范围内
265 if (value < 0 || value > 1000)
266 {
267 std::cerr << "Error: Value out of range [0, 1000]" << std::endl;
268 return;
269 }
270
271 // 将 value 转换为十六进制并存储到 data 数组中
272 std::stringstream ss;
273 ss << std::hex << std::uppercase << std::setw(4) << std::setfill('0') << value;
274 std::string hexString = ss.str();
275
276 // 检查是否需要补充 0，确保有 4 位
277 if (hexString.length() < 4)
278 {
279 hexString = std::string(4 - hexString.length(), '0') + hexString;
280 }
281
282 // 交换高低字节
283 data[0] = std::stoi(hexString.substr(2, 2), nullptr, 16); // 高字节
284 data[1] = std::stoi(hexString.substr(0, 2), nullptr, 16); // 低字节
285}
286int Inspire::WriteCan(int address, int value, int handid)
287{
288 std::string canid = GetCanId(address, handid, 1);
289 if (canid.size() != 32)
290 {
291 std::cout << "canid is error" << std::endl;
292 return -1;
293 }
294 // std::cout << "canid:" << canid << std::endl;
295 unsigned int Canid = std::bitset<32>(canid).to_ulong();
296
297 // frame.ID = 0000 0101 0111 0101 0000 0000 0000 0001
298
299 Byte data[2];
300 ConvertValueToHex(value, data);
301 frame.ID = Canid;
302
303 memcpy(frame.Data, data, frame.DataLen);
304 if (Transmit(USBCAN2, DeviceInd, CAN1, &frame, 1) != STATUS_OK)

306 std::cout << "it is error when transmit" << std::endl;
307 CloseDevice(USBCAN2, DeviceInd);
